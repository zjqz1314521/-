#include <reg52.h>

sbit KEY = P3^5; //定义 增强可读性
int i;     //定义要在前

void main(void)
{
	TMOD = 0x20;  //为什么定时器模式要是0x20？0x寄存器的算法为16进制，2为0010 0为0000 前四位为寄存器1后四位为寄存器0 由前向后解释 0允许不启动外部中断功能运行 0设置定时器模式 10 mode2 8位自动重装载定时器 00 
	//mode 0 13位定时器 低电平5位高电平8位
	TH1 = 0xFD; 		//确认11.0592M下的值 FD换算为二进制 1111 1101 换算为十进制为253 即计数3位则达到255 溢出 11.0592/12/3 =0.3072MHz  
	TL1 = 0xFD;
	TR1 = 1;       //启动定时器
	
	SCON = 0x50;  //SM0和SM1组合什么意思 0101 1000 模式
	
	while(1){
		if(KEY == 0){   //软件防误抖？感觉可以
			unsigned int i;
			for(i=0; i<5000; i++);
			
			if(KEY == 0){  
			P1 = 0x00;     //全亮，设备0为亮，1为暗
			
			SBUF = 0x55;     //SBUF符号0101 0101是U
			while(!TI);      //等发送完成
			TI = 0;          //完成后清除
		  
			while(KEY == 0);  //未松按键时，循环，，，，测试当KEY等于1没啥用了这玩意，led频闪，U字符连发
				P1 = 0xFF;
			}
		}
		
	}
}
